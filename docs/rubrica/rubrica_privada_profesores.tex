\textbf{Criterios de correccion} \hfill \textbf{Total: 10 puntos}

\begin{multicols}{2}

\bloque{UML}{3 puntos}
\subbloque{Relaciones}{0.5}
\begin{itemize}
    \item Falta una relacion o su tipo es incorrecto: -0.15 por caso.
    \item Faltan multiplicidades: -0.10 por caso.
\end{itemize}
\subbloque{Clases}{0.5}
\begin{itemize}
    \item Falta una clase relevante: -0.20 por caso.
    \item Aparece una clase que no corresponde: -0.10.
    \item Tipo de elemento UML incorrecto: -0.20.
\end{itemize}
\subbloque{Contenido}{2}
\begin{itemize}
    \item Faltan atributos: -0.25.
    \item Faltan metodos principales: -0.25.
    \item No se indican visibilidades: -0.25.
    \item Faltan tipos, parametros o retornos: -0.25.
    \item No se reflejan excepciones propias: -0.20.
\end{itemize}

\bloque{Main}{1 punto}
\begin{itemize}
    \item Fallo en la creacion y/o invocacion de objetos: -0.45.
    \item Main monolitico sin funciones auxiliares cuando son necesarias: -0.75.
    \item Main contiene logica de dominio: -0.30.
\end{itemize}

\bloque{Implementacion general}{5 puntos}
\subbloque{Navegacion por menus}{0.60}
\begin{itemize}
    \item Acceso claro a las opciones principales.
    \item Navegacion funcional entre menus y opciones.
    \item Delegacion en clases del sistema.
\end{itemize}
\subbloque{Equipar y desequipar}{1.00}
\begin{itemize}
    \item Equipar y desequipar correctamente.
    \item Inventario, equipo, ataque y defensa coherentes.
    \item Uso correcto de interfaz y polimorfismo.
    \item Respeto de la encapsulacion.
\end{itemize}
\subbloque{Comprar en tienda}{0.80}
\begin{itemize}
    \item Compra funcional.
    \item Tienda, inventario y dinero actualizados.
    \item Informacion de objetos conservada, incluido metal.
    \item Encapsulacion entre menu, tienda, personaje e inventario.
\end{itemize}
\subbloque{Guardar personaje}{0.50}
\begin{itemize}
    \item Guarda el estado actual.
    \item Conserva informacion principal, equipo e inventario.
    \item Formato coherente con la lectura.
\end{itemize}
\subbloque{Uso de \texttt{toString}}{0.40}
\begin{itemize}
    \item Representacion textual util.
    \item Uso desde el menu para mostrar informacion.
    \item Salida clara de personaje, inventario, equipo y tienda.
\end{itemize}
\subbloque{Herencia de \texttt{Objeto}}{0.90}
\begin{itemize}
    \item \texttt{Objeto} como clase base.
    \item Herencia correcta en arma y armadura.
    \item No duplicar atributos o logica comun.
    \item Constructores y comportamiento comun bien planteados.
\end{itemize}
\subbloque{Uso de \texttt{enum}}{0.20}
\begin{itemize}
    \item Uso cuando hay conjunto cerrado de valores.
    \item Integracion real en la logica.
\end{itemize}
\subbloque{Encapsulacion}{0.60}
\begin{itemize}
    \item Datos internos protegidos.
    \item Responsabilidades en la clase adecuada.
    \item Sin accesos directos a arrays internos desde fuera.
    \item Metodos de dominio en lugar de manipular estructuras internas.
\end{itemize}

\bloque{Excepciones}{1 punto}
\begin{itemize}
    \item Faltan excepciones propias necesarias: -0.25 por cada una.
    \item Excepcion no lanzada en sitio adecuado: -0.10 por caso.
    \item Excepcion no capturada en sitio adecuado: -0.10 por caso.
    \item Captura generica que oculta errores o vacia el flujo: -0.10.
    \item No gestiona bien entrada/salida: -0.20.
\end{itemize}

\bloque{Penalizaciones globales}{ }
\begin{itemize}
    \item UML no corresponde con el codigo: -2.00.
    \item Codigo comentado, muerto o de pruebas: -1.00.
    \item Nombres o formato muy deficientes: -0.50.
    \item Mas de un return en una funcion: -1.00. Pendiente decidir alcance.
    \item Break fuera de switch: -1.00. Pendiente decidir alcance.
    \item Continue: -1.00. Pendiente decidir alcance.
\end{itemize}

\bloque{Techos de nota}{ }
\begin{itemize}
    \item No compila: maximo 3.00.
    \item Compila pero no permite ejecutar un flujo basico: maximo 3.00.
    \item Usa estructuras explicitamente prohibidas: maximo 3.00.
    \item No supera la defensa: maximo 3.00.
\end{itemize}

Un techo no asigna directamente esa calificacion: se corrige sobre ese maximo y las penalizaciones siguen pudiendo restar dentro de ese rango.

\end{multicols}
